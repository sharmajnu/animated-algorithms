int binarySearch(int[] array, int number)
{
       start = 0;
       end = array.length - 1;
       while(start <= end)
       {
              mid = start + (end-start)/2;
              if(array[mid] == number)
                      return mid;
              else if(number < array[mid])
                     end = mid - 1;
              else
                     start = mid + 1;
       }
       return -1;
}
